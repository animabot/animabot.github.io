654  typename InputImageType::SpacingType fixedSpacing = m_BackwardImage->GetSpacing();
655 
656  LocalOptimizerType::ScalesType tmpScales(m_BaseTransformsPointers[startBlock]->GetNumberOfParameters());
657  LocalOptimizerType::ScalesType lowerBounds(m_BaseTransformsPointers[startBlock]->GetNumberOfParameters());
658  LocalOptimizerType::ScalesType upperBounds(m_BaseTransformsPointers[startBlock]->GetNumberOfParameters());
659 
660  // Scale factor to ensure that max translations and skew can be reached
661  // Based on the fact that non diagonal terms log is a = x * log(y) / (exp(y) - 1)
662  // where y is the diagonal scaling factor, x the desired term
663  double scaleFactor = 1.0;
664  if ((m_TransformationKind != Direction)&&(m_ScaleMax > 0))
665  scaleFactor = - m_ScaleMax / (std::exp(- m_ScaleMax) - 1.0);
666 
667  switch (m_TransformationKind)
668  {
669  case DirectionScaleSkew:
670  {
671  tmpScales[0] = m_SearchRadius / m_SearchScaleRadius;
672  tmpScales[1] = m_SearchRadius / m_SearchSkewRadius;
673  tmpScales[2] = m_SearchRadius / m_SearchSkewRadius;
674  tmpScales[3] = 1.0;
675 
676  lowerBounds[0] = - m_ScaleMax;
677  upperBounds[0] = m_ScaleMax;
678  lowerBounds[1] = - m_SkewMax * scaleFactor;
679  upperBounds[1] = m_SkewMax * scaleFactor;
680  lowerBounds[2] = - m_SkewMax * scaleFactor;
681  upperBounds[2] = m_SkewMax * scaleFactor;
682  lowerBounds[3] = - m_TranslateMax * scaleFactor;
683  upperBounds[3] = m_TranslateMax * scaleFactor;
684 
685  break;
686  }
687 
688  case DirectionScale:
689  {
690  tmpScales[0] = m_SearchRadius / m_SearchScaleRadius;
691  tmpScales[1] = 1.0;
692 
693  lowerBounds[0] = - m_ScaleMax;
694  upperBounds[0] = m_ScaleMax;
695  lowerBounds[1] = - m_TranslateMax * scaleFactor;
696  upperBounds[1] = m_TranslateMax * scaleFactor;
697 
698  break;
699  }
700 
701 
702  case Direction:
703  default:
704  {
705  tmpScales[0] = 1.0;
706  lowerBounds[0] = - m_TranslateMax;
707  upperBounds[0] = m_TranslateMax;
708 
709  break;
710  }
711  }
712 
713  tmpOpt->SetScales(tmpScales);
714  tmpOpt->SetLowerBounds(lowerBounds);
715  tmpOpt->SetUpperBounds(upperBounds);
716  tmpOpt->SetMaximize(m_MaximiseMetric);
717  metric->ComputeGradientOff();
718 }

721 void
722 DistortionCorrectionRegistrationMethod<TInputImage>
723 ::BlockMatch(unsigned threadId, unsigned NbThreads, InputImageType *fixedImage,
724  InputImageType *movingImage)
725 {
726  InterpolatorPointer interpolator;
727  MetricPointer metric;
728  OptimizerPointer optimizer;
729 
730  unsigned totalRegionSize = m_BlockRegions.size();
731  unsigned step = totalRegionSize / NbThreads;
732 
733  unsigned startBlock = threadId*step;
734  unsigned endBlock = (1+threadId)*step;
735 
736  if (threadId+1==NbThreads) // Ensure we got up to the last block
737  endBlock = totalRegionSize;
738 
739  setUpParameters(interpolator, metric, optimizer, startBlock, endBlock);
740  metric->SetFixedImage(fixedImage);
741  metric->SetMovingImage(movingImage);
742 
743  if (GetDebug())
744  {
745  CommandIterationUpdate::Pointer observer = CommandIterationUpdate::New();
746  optimizer->AddObserver( itk::IterationEvent(), observer );
747  }
748 
749  itkDebugMacro(<<"Threaded Block Match " << threadId << " / " << NbThreads << " starting @ " << startBlock << " up to " << endBlock );
750 
751  // Loop over the desired blocks
752  typename BaseInputTransformType::Pointer realTransform;
753  for (unsigned int block = startBlock; block < endBlock; ++block)
754  {
755  metric->SetFixedImageRegion(m_BlockRegions[block]);
756  metric->SetTransform(m_BaseTransformsPointers[block]);
757  metric->Initialize();
758 
759  optimizer->SetCostFunction(metric);
760  optimizer->SetInitialPosition(m_InitialTransformParameters);
761 
762  if (m_MetricKind != MeanSquares)
763  ((anima::FastCorrelationImageToImageMetric<TInputImage, TInputImage> *)metric.GetPointer())->PreComputeFixedValues();
764  else
765  ((anima::FastMeanSquaresImageToImageMetric<TInputImage, TInputImage> *)metric.GetPointer())->PreComputeFixedValues();
766 
767  try
768  {
769  optimizer->StartOptimization();
770  }
771  catch( itk::ExceptionObject & err )
772  {
773  std::cerr << "ExceptionObject caught Image Block Match! " << err << std::endl;
774  m_BlockWeights[block] = 0;
775  continue;
776  }
777 
778  m_BaseTransformsPointers[block]->SetParameters( optimizer->GetCurrentPosition() );
779 
780  if (m_WeightedAgregation && (m_MetricKind != MeanSquares))
781  {
782  double val = ((anima::BobyqaOptimizer *)optimizer.GetPointer())->GetValue();
783  m_BlockWeights[block] = (val >= 0) ? val : -val;
784  }
785  else
786  m_BlockWeights[block] = 1;
787 
788  m_BlockWeights[block] *= this->ComputeDirectionFilterBlockWeightUpdate(fixedImage,m_BlockRegions[block]);
789  m_BlockWeights[block] = std::pow(m_BlockWeights[block],1/2.0);
790 
791  itkDebugMacro(<< m_BaseTransformsPointers[block]->GetParameters() << " " << m_BlockWeights[block] << std::endl);
792  }

795 template <typename TInputImage>
796 double
797 DistortionCorrectionRegistrationMethod<TInputImage>
798 ::ComputeDirectionFilterBlockWeightUpdate(InputImageType *fixedImage, const ImageRegionType &region)
799 {
800  std::vector <double> localGradient(TInputImage::ImageDimension,0);
801  itk::ImageRegionConstIterator <InputImageType> fixedItr(fixedImage,region);
802  typename ImageRegionType::IndexType currentIndex, modifiedIndex;
803 
804  typename InputImageType::DirectionType orientationMatrix = fixedImage->GetDirection();
805  typename InputImageType::SpacingType imageSpacing = fixedImage->GetSpacing();
806  typename InputImageType::SizeType imageSize = fixedImage->GetLargestPossibleRegion().GetSize();
807 
808  std::vector <double> correctionDirection(TInputImage::ImageDimension);
809  for (unsigned int i = 0;i < TInputImage::ImageDimension;++i)
810  correctionDirection[i] = fixedImage->GetDirection()(i,m_TransformDirection);
811 
812  vnl_matrix <double> meanStructureTensor(TInputImage::ImageDimension,TInputImage::ImageDimension);
813  meanStructureTensor.fill(0);
814 
815  while (!fixedItr.IsAtEnd())
816  {
817  currentIndex = fixedItr.GetIndex();
818  std::fill(localGradient.begin(),localGradient.end(),0);
819 
820  for (unsigned int i = 0;i < TInputImage::ImageDimension;++i)
821  {
822  modifiedIndex = currentIndex;
823  modifiedIndex[i] = std::max(0,(int)(currentIndex[i] - 1));
824  double previousValue = fixedImage->GetPixel(modifiedIndex);
825  modifiedIndex[i] = std::min((int)(imageSize[i] - 1),(int)(currentIndex[i] + 1));
826  double postValue = fixedImage->GetPixel(modifiedIndex);
827 
828  for (unsigned int j = 0;j < TInputImage::ImageDimension;++j)
829  localGradient[j] += (postValue - previousValue) * orientationMatrix(j,i) / (2.0 * imageSpacing[i]);
830  }
831 
832  for (unsigned int i = 0;i < TInputImage::ImageDimension;++i)
833  for (unsigned int j = i;j < TInputImage::ImageDimension;++j)
834  {
835  meanStructureTensor(i,j) += localGradient[i] * localGradient[j];
836  if (j != i)
837  meanStructureTensor(j,i) = meanStructureTensor(i,j);
838  }
839 
840  ++fixedItr;
841  }
842 
843  meanStructureTensor /= region.GetNumberOfPixels();
844 
845  itk::SymmetricEigenAnalysis < vnl_matrix <double>, vnl_diag_matrix<double>, vnl_matrix <double> > eigenComputer(TInputImage::ImageDimension);
846  vnl_matrix <double> eVec(TInputImage::ImageDimension,TInputImage::ImageDimension);
847  vnl_diag_matrix <double> eVals(TInputImage::ImageDimension);
848 
849  eigenComputer.ComputeEigenValuesAndVectors(meanStructureTensor, eVals, eVec);
850  double linearCoef = (eVals[TInputImage::ImageDimension - 1] - eVals[TInputImage::ImageDimension - 2]) / eVals[TInputImage::ImageDimension - 1];
851 
852  double scalarProduct = 0;
853  for (unsigned int i = 0;i < TInputImage::ImageDimension;++i)
854  scalarProduct += eVec[TInputImage::ImageDimension - 1][i] * correctionDirection[i];
855 
856  return linearCoef * std::abs(scalarProduct);
857  //return std::abs(scalarProduct);
858 }
