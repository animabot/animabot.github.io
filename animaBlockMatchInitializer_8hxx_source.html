401 template <class PixelType, unsigned int NDimensions>
402 bool BlockMatchingInitializer<PixelType,NDimensions>::
403 CheckTensorVariance(VectorImageType *refImage, ImageRegionType &region, double &blockVariance)
404 {
405  itk::ImageRegionConstIterator <VectorImageType> refItr(refImage,region);
406  typedef typename VectorImageType::PixelType VectorType;
407 
408  unsigned int nbPts = 0;
409  unsigned int vectorSize = refImage->GetNumberOfComponentsPerPixel();
410 
411  VectorType meanVal(vectorSize);
412  meanVal.Fill(0);
413  vnl_matrix <double> blockCovariance(vectorSize,vectorSize);
414  blockCovariance.fill(0);
415 
416  while (!refItr.IsAtEnd())
417  {
418  VectorType tmpVal = refItr.Get();
419  meanVal += tmpVal;
420 
421  for (unsigned int j = 0;j < vectorSize;++j)
422  for (unsigned int k = j;k < vectorSize;++k)
423  blockCovariance(j,k) += tmpVal[j] * tmpVal[k];
424 
425  ++nbPts;
426 
427  ++refItr;
428  }
429 
430  if (nbPts <= 1)
431  return false;
432 
433  blockVariance = 0;
434 
435  for (unsigned int j = 0;j < vectorSize;++j)
436  for (unsigned int k = j;k < vectorSize;++k)
437  {
438  double tmp = (blockCovariance(j,k) - meanVal[j]*meanVal[k]/nbPts)/(nbPts - 1.0);
439  if (j == k)
440  blockVariance += tmp * tmp;
441  else
442  blockVariance += 2 * tmp * tmp;
443  }
444 
445  blockVariance = sqrt(blockVariance);
446 
447  if (blockVariance > this->GetTensorVarianceThreshold())
448  return true;
449  else
450  return false;
451 }
