325 template <typename TInputImage>
326 bool
327 TensorBlockMatchingBaseRegistrationMethod<TInputImage>
328 ::ComposeAddOnWithTransform(TransformType *computedTransform, TransformType *addOn)
329 {
330  if (m_Agregator->GetOutputTransformType() != AgregatorType::SVF)
331  {
332  typename TransformType::ParametersType oldPars = computedTransform->GetParameters();
333 
334  AffineTransformType *tmpTrsf = dynamic_cast<AffineTransformType *>(computedTransform);
335  AffineTransformType *tmpAddOn = dynamic_cast<AffineTransformType *>(addOn);
336  tmpTrsf->Compose(tmpAddOn, true);
337 
338  typename TransformType::ParametersType newPars = computedTransform->GetParameters();
339 
340  // Compute the distance between consecutive solutions, until a certain threshold
341  double err = 0;
342  for (unsigned int i = 0; i < newPars.Size(); ++i)
343  err += pow(newPars[i] - oldPars[i], 2.);
344 
345  if (err < m_MinimalTransformError)
346  {
347  m_StopConditionDescription << "Minimal Transform Error Reached " << err;
348  return false;
349  }
350  }
351  else
352  {
353  // Add update to current velocity field (cf. Vercauteren et al, 2008)
354  SVFTransformType *tmpTrsf = dynamic_cast<SVFTransformType *>(computedTransform);
355  SVFTransformType *tmpAddOn = dynamic_cast<SVFTransformType *>(addOn);
356 
357  anima::composeSVF(tmpTrsf,tmpAddOn);
358  if (m_SVFElasticRegSigma > 0)
359  {
360  typedef typename SVFTransformType::VectorFieldType VelocityFieldType;
361 
362  typedef anima::SmoothingRecursiveYvvGaussianImageFilter <VelocityFieldType, VelocityFieldType> SmoothingFilterType;
363  typename SmoothingFilterType::Pointer smootherPtr = SmoothingFilterType::New();
364 
365  smootherPtr->SetInput(tmpTrsf->GetParametersAsVectorField());
366  smootherPtr->SetSigma(m_SVFElasticRegSigma);
367  smootherPtr->SetNumberOfThreads(this->GetNumberOfThreads());
368 
369  smootherPtr->Update();
370 
371  typename VelocityFieldType::Pointer tmpSmoothed = smootherPtr->GetOutput();
372  tmpSmoothed->DisconnectPipeline();
373 
374  tmpTrsf->SetParametersAsVectorField(tmpSmoothed);
375  }
376  }
377 
378  return true;

726 template <typename TInputImage>
727 void
728 TensorBlockMatchingBaseRegistrationMethod<TInputImage>
729 ::BlockMatch(unsigned threadId, unsigned NbThreads, InputImageType *fixedImage,
730  InputImageType *movingImage)
731 {
732  InterpolatorPointer interpolator;
733  MetricPointer metric;
734  OptimizerPointer optimizer;
735 
736  unsigned totalRegionSize = m_BlockRegions.size();
737  unsigned step = totalRegionSize / NbThreads;
738 
739  unsigned startBlock = threadId*step;
740  unsigned endBlock = (1+threadId)*step;
741 
742  if (threadId+1==NbThreads) // Ensure we got up to the last block
743  endBlock = totalRegionSize;
744 
745  this->SetupBlockParameters(interpolator, metric, optimizer, startBlock, endBlock);
746 
747  metric->SetFixedImage(fixedImage);
748  metric->SetMovingImage(movingImage);
749  interpolator->SetInputImage(movingImage);
750 
751  itkDebugMacro(<<"Threaded Block Match " << threadId << " / " << NbThreads << " starting @ " << startBlock << " up to " << endBlock );
752 
753  // Loop over the desired blocks
754  typename BaseInputTransformType::Pointer realTransform;
755  for (unsigned int block = startBlock; block < endBlock; ++block)
756  {
757 
758  metric->SetFixedImageRegion(m_BlockRegions[block]);
759  metric->SetTransform(m_BaseTransformsPointers[block]);
760  metric->Initialize();
761 
762  this->AdditionalBlockMatchingSetup(metric,block);
763 
764  optimizer->SetCostFunction(metric);
765  optimizer->SetInitialPosition(m_InitialTransformParameters);
766 
767  try
768  {
769  optimizer->StartOptimization();
770  }
771  catch( itk::ExceptionObject & err )
772  {
773  std::cout << "ExceptionObject caught Image Block Match! " << err << std::endl;
774  m_BlockWeights[block] = 0;
775  continue;
776  }
777 
778  m_BaseTransformsPointers[block]->SetParameters(optimizer->GetCurrentPosition());
779 
780  if (m_MetricKind != TensorMeanSquares)
781  {
782  double val = 0;
783  switch (m_OptimizerKind)
784  {
785  case Bobyqa:
786  val = ((anima::BobyqaOptimizer *)optimizer.GetPointer())->GetValue();
787  break;
788 
789  case Exhaustive:
790  default:
791  if (m_MaximizeMetric)
792  val = ((anima::VoxelExhaustiveOptimizer *)optimizer.GetPointer())->GetCurrentCost();
793 
794  break;
795  }
796 
797  m_BlockWeights[block] = (val >= 0) ? val : -val;
798  }
799  else
800  m_BlockWeights[block] = 1;
801 
802  itkDebugMacro(<< m_BaseTransformsPointers[block]->GetParameters() << " " << m_BlockWeights[block] << std::endl);
803  }
804 }
