330 template <typename TInputImage>
331 bool
332 BlockMatchingBaseRegistrationMethod<TInputImage>
333 ::ComposeAddOnWithTransform(TransformType *computedTransform, TransformType *addOn)
334 {
335  if (m_Agregator->GetOutputTransformType() != AgregatorType::SVF)
336  {
337  typename TransformType::ParametersType oldPars = computedTransform->GetParameters();
338 
339  AffineTransformType *tmpTrsf = dynamic_cast<AffineTransformType *>(computedTransform);
340  AffineTransformType *tmpAddOn = dynamic_cast<AffineTransformType *>(addOn);
341  tmpTrsf->Compose(tmpAddOn, true);
342 
343  typename TransformType::ParametersType newPars = computedTransform->GetParameters();
344 
345  // Compute the distance between consecutive solutions, until a certain threshold
346  double err = 0;
347  for (unsigned int i = 0; i < newPars.Size(); ++i)
348  err += pow(newPars[i] - oldPars[i], 2.);
349 
350  if (err < m_MinimalTransformError)
351  {
352  m_StopConditionDescription << "Minimal Transform Error Reached " << err;
353  return false;
354  }
355  }
356  else
357  {
358  // Add update to current velocity field (cf. Vercauteren et al, 2008)
359  SVFTransformType *tmpTrsf = dynamic_cast<SVFTransformType *>(computedTransform);
360  SVFTransformType *tmpAddOn = dynamic_cast<SVFTransformType *>(addOn);
361 
362  anima::composeSVF(tmpTrsf,tmpAddOn);
363  if (m_SVFElasticRegSigma > 0)
364  {
365  typedef typename SVFTransformType::VectorFieldType VelocityFieldType;
366 
367  typedef anima::SmoothingRecursiveYvvGaussianImageFilter <VelocityFieldType, VelocityFieldType> SmoothingFilterType;
368  typename SmoothingFilterType::Pointer smootherPtr = SmoothingFilterType::New();
369 
370  smootherPtr->SetInput(tmpTrsf->GetParametersAsVectorField());
371  smootherPtr->SetSigma(m_SVFElasticRegSigma);
372  smootherPtr->SetNumberOfThreads(this->GetNumberOfThreads());
373 
374  smootherPtr->Update();
375 
376  typename VelocityFieldType::Pointer tmpSmoothed = smootherPtr->GetOutput();
377  tmpSmoothed->DisconnectPipeline();
378 
379  tmpTrsf->SetParametersAsVectorField(tmpSmoothed);
380  }
381  }
382 
383  return true;

709 template <typename TInputImage>
710 void
711 BlockMatchingBaseRegistrationMethod<TInputImage>
712 ::BlockMatch(unsigned threadId, unsigned NbThreads, InputImageType *fixedImage,
713  InputImageType *movingImage)
714 {
715  InterpolatorPointer interpolator;
716  MetricPointer metric;
717  OptimizerPointer optimizer;
718 
719  unsigned totalRegionSize = m_BlockRegions.size();
720  unsigned step = totalRegionSize / NbThreads;
721 
722  unsigned startBlock = threadId*step;
723  unsigned endBlock = (1+threadId)*step;
724 
725  if (threadId+1==NbThreads) // Ensure we got up to the last block
726  endBlock = totalRegionSize;
727 
728  this->SetupBlockParameters(interpolator, metric, optimizer, startBlock, endBlock);
729 
730  metric->SetFixedImage(fixedImage);
731  metric->SetMovingImage(movingImage);
732  interpolator->SetInputImage(movingImage);
733 
734  itkDebugMacro(<<"Threaded Block Match " << threadId << " / " << NbThreads << " starting @ " << startBlock << " up to " << endBlock );
735 
736  // Loop over the desired blocks
737  typename BaseInputTransformType::Pointer realTransform;
738  for (unsigned int block = startBlock; block < endBlock; ++block)
739  {
740 
741  metric->SetFixedImageRegion(m_BlockRegions[block]);
742  metric->SetTransform(m_BaseTransformsPointers[block]);
743  metric->Initialize();
744 
745  this->AdditionalBlockMatchingSetup(metric, block);
746 
747  optimizer->SetCostFunction(metric);
748  optimizer->SetInitialPosition(m_InitialTransformParameters);
749 
750  try
751  {
752  optimizer->StartOptimization();
753  }
754  catch( itk::ExceptionObject & err )
755  {
756  std::cout << "ExceptionObject caught Image Block Match! " << err << std::endl;
757  m_BlockWeights[block] = 0;
758  continue;
759  }
760 
761  m_BaseTransformsPointers[block]->SetParameters(optimizer->GetCurrentPosition());
762 
763  if (m_MetricKind != MeanSquares)
764  {
765  double val = 0;
766  switch (m_OptimizerKind)
767  {
768  case Bobyqa:
769  val = ((anima::BobyqaOptimizer *)optimizer.GetPointer())->GetValue();
770  break;
771 
772  case Exhaustive:
773  default:
774  if (m_MaximizeMetric)
775  val = ((anima::VoxelExhaustiveOptimizer *)optimizer.GetPointer())->GetCurrentCost();
776 
777  break;
778  }
779 
780  m_BlockWeights[block] = (val >= 0) ? val : -val;
781  }
782  else
783  m_BlockWeights[block] = 1;
784 
785  itkDebugMacro(<< m_BaseTransformsPointers[block]->GetParameters() << " " << m_BlockWeights[block] << std::endl);
786  }
787 }
