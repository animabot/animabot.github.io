483

485 
486  vnl_matrix <double> tmpMat(3,3);
487 
488  unsigned int pos = 0;
489  for (unsigned int i = 0;i < 3;++i)
490  for (unsigned int j = 0;j <= i;++j)
491  {
492  tmpMat(i,j) = dtiLogValue[pos];
493  if (j != i)
494  tmpMat(j,i) = tmpMat(i,j);
495  ++pos;
496  }
497 
498  vnl_diag_matrix <double> eVals(3);
499  EigenAnalysis.ComputeEigenValues(tmpMat,eVals);
500 
501  double meanEvals = 0;
502  for (unsigned int i = 0;i < 3;++i)
503  {
504  eVals[i] = exp(eVals[i]);
505  meanEvals += eVals[i];
506  }
507 
508  meanEvals /= 3.0;
509 
510  double num = 0;
511  double denom = 0;
512  for (unsigned int i = 0;i < 3;++i)
513  {
514  num += (eVals[i] - meanEvals) * (eVals[i] - meanEvals);
515  denom += eVals[i] * eVals[i];
516  }
517 
518  return sqrt(num / (2 * denom));
