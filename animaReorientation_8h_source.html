142 reorientImageAndGradient(typename itk::SmartPointer<ImageType> input,
143  itk::SpatialOrientation::ValidCoordinateOrientationFlags orientation,
144  std::vector<GradientType> &gradients)
145 {
146  // What we want to do here is reorient an image and then a list of gradient
147  // vectors in order to have them in real coordinates, oriented in the
148  // same way that the newly oriented image.
149 
150  // To do so we apply 'reorientedGrad = Dr o grad', with: 'Dr = D1^-1 o D0'
151 
152  // grad are the gradients oriented as given in input (in real coordinates repair of input image).
153  // D0 is the direction matrix of the image given as input to translate image coordinate in real coordinates
154  // D1 is the direction matrix of the reoriented image to translate image coordinate in real coordinates
155 
156  typename ImageType::Pointer reorientedImage = reorientImage<ImageType>(input, orientation);
157 
158  typename ImageType::DirectionType D0 = input->GetDirection();
159  typename ImageType::DirectionType inversedD1 = reorientedImage->GetInverseDirection();
160 
161  typename ImageType::DirectionType Dr;
162  Dr.Fill(0);
163 
164  for(unsigned int i = 0; i < 3; ++i)
165  for(unsigned int j = 0; j < 3; ++j)
166  for(unsigned int n = 0; n < 3; ++n)
167  Dr[i][j] += inversedD1[i][n] * D0[n][j];
168 
169  for(unsigned int g = 0; g < gradients.size(); g++)
170  {
171  GradientType reorientedGrad(3,0);
172  for(unsigned int i = 0; i < 3; ++i)
173  {
174  for(unsigned int j = 0; j < 3; ++j)
175  reorientedGrad[i] += Dr[i][j] * gradients[g][j];
176  }
