237 template <class T1, class VectorType> double ComputeSymmetricCDF(const VectorType &direction, const T1 &kappa, const T1 &lambda, const T1 &nu,
238  const T1 &bvalue, const VectorType &gradient)
239 {
240  double resVal;
241 
242  double vmfLambda = nu * lambda;
243  double gaussLambda = (1.0 - nu) * lambda;
244 
245  double muG = anima::ComputeScalarProduct(direction, gradient);
246 
247  resVal = std::exp( -bvalue * gaussLambda * (1 + kappa * muG * muG) / (kappa + 1) );
248 
249  double sqrt2bLambda = std::sqrt(2.0 * bvalue * vmfLambda);
250  double kappaSq = kappa * kappa;

252  double ReZ = kappaSq - sqrt2bLambda * sqrt2bLambda;
253  double ImZ = 2.0 * kappa * sqrt2bLambda * muG;

255  double condition = ReZ + std::sqrt(ReZ*ReZ + ImZ*ImZ);
256  if (condition < 0)

259

265  }

269  double beta = ImZ / std::sqrt(2.0 * condition);

271  resVal *= tmp;
