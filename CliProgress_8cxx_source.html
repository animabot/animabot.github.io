23  void encodeblock(const unsigned char in[3], unsigned char out[4], int len )
24  {
25  static const char cb64[]="ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/";
26 
27  out[0] = cb64[ in[0] >> 2 ];
28  if (len == 1)
29  out[1] = cb64[ ((in[0] & 0x03) << 4) ];
30  else
31  out[1] = cb64[ ((in[0] & 0x03) << 4) | ((in[1] & 0xf0) >> 4) ];
32  if (len > 1)
33  out[2] = (unsigned char) (len > 1 ? cb64[ ((in[1] & 0x0f) << 2) | ((in[2] & 0xc0) >> 6) ] : '=');
34  if (len > 2)
35  out[3] = (unsigned char) (len > 2 ? cb64[ in[2] & 0x3f ] : '=');
36  }
