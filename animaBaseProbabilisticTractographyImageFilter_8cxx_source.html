663  }

669  logWeightSums[i] = 0;
670  emptyClasses[i] = true;
671  }

696  }

702  {
703  if (fiberComputationData.stoppedParticles[i])

728  {

730  effectiveNumberOfParticles[m] = 1.0 / effectiveNumberOfParticles[m];
731  else
732  continue; // Q: shouldn't we be treating this case as an empty cluster that shouldn't even exist? (same as previous Q)

735  if (effectiveNumberOfParticles[m] < m_ResamplingThreshold * fiberComputationData.classSizes[m])
736  {
737  weightSpecificClassValues.resize(fiberComputationData.classSizes[m]);
738  previousDirectionsCopy.resize(fiberComputationData.classSizes[m]);
739  fiberParticlesCopy.resize(fiberComputationData.classSizes[m]);

742  {

744  previousDirectionsCopy[i] = previousDirections[fiberComputationData.reverseClassMemberships[m][i]];
745  fiberParticlesCopy[i] = fiberComputationData.fiberParticles[fiberComputationData.reverseClassMemberships[m][i]];

762  }

767

772  if (fiberComputationData.particleWeights[iReal] != 0)
773  fiberParticlesCopy[i].pop_back();
774 
775  // The fiber trash used to contain fibers that were lost with a sufficient weight

778  // fiberTrash.push_back(fiberParticlesCopy[i]);
779  }
780  }
781  }
782

787  }
788  }

791  // Length is easy, given that each step is constant we just need to check the fiber size: numIter
792  // Example :

794  stopLoop = true;
795 
796  numberOfClasses = this->UpdateClassesMemberships(fiberComputationData,previousDirections);
797 
798  for (unsigned int i = 0;i < fiberComputationData.particleWeights.size();++i)
799  {
800  if (!boost::math::isfinite(fiberComputationData.particleWeights[i]))
801  itkExceptionMacro("Nan weights after update class membership");
802  }
803  }
804

807  {
808  FiberProcessVectorType mergedOutput,classMergedOutput;
809  for (unsigned int i = 0;i < numberOfClasses;++i)

820  else
821  resultWeights = fiberComputationData.particleWeights;
822 
823  return fiberComputationData.fiberParticles;
