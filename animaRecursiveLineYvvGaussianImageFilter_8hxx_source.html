57 void
58 RecursiveLineYvvGaussianImageFilter<TInputImage,TOutputImage>
59 ::SetUp(ScalarRealType spacing)
60 {
61  const ScalarRealType sigmad = m_Sigma/spacing;
62 
63  // Compute q according to 16 in Young et al on Gabor filering
64  ScalarRealType q = 0;
65  if (sigmad >= 3.556)
66  q = 0.9804 * (sigmad - 3.556) + 2.5091;
67  else
68  {
69  if (sigmad < 0.5)
70  std::cerr << "Too low sigma value (< 0.5), computation will not be precise." << std::endl;
71 
72  q = 0.0561 * sigmad * sigmad + 0.5784 * sigmad - 0.2568;
73  }
74 
75  // Compute B and B1 to B3 according to Young et al 2003
76  ScalarRealType m0 = 1.16680;
77  ScalarRealType m1 = 1.10783;
78  ScalarRealType m2 = 1.40586;
79  ScalarRealType scale = (m0 + q) * (m1 * m1 + m2 * m2 + 2 * m1 * q + q * q);
80 
81  m_B1 = q * (2 * m0 * m1 + m1 * m1 + m2 * m2 + (2 * m0 + 4 * m1) * q + 3 * q * q) / scale;
82 
83  m_B2 = - q * q * (m0 + 2 * m1 + 3 * q) / scale;
84 
85  m_B3 = q * q * q / scale;
86 
87  ScalarRealType baseB = (m0 * (m1 * m1 + m2 * m2)) / scale;
88  m_B = baseB * baseB;
89 
90  // M Matrix for initialization on backward pass, from Triggs and Sdika, IEEE TSP
91  m_MMatrix = vnl_matrix <ScalarRealType> (3,3);
92 
93  m_MMatrix(0,0) = - m_B3 * m_B1 + 1 - m_B3 * m_B3 - m_B2;
94  m_MMatrix(0,1) = (m_B3 + m_B1) * (m_B2 + m_B3 * m_B1);
95  m_MMatrix(0,2) = m_B3 * (m_B1 + m_B3 * m_B2);
96 
97  m_MMatrix(1,0) = m_B1 + m_B3 * m_B2;
98  m_MMatrix(1,1) = (1 - m_B2) * (m_B2 + m_B3 * m_B1);
99  m_MMatrix(1,2) = - m_B3 * (m_B3 * m_B1 + m_B3 * m_B3 + m_B2 - 1);
100 
101  m_MMatrix(2,0) = m_B3 * m_B1 + m_B2 + m_B1 * m_B1 - m_B2 * m_B2;
102  m_MMatrix(2,1) = m_B1 * m_B2 + m_B3 * m_B2 * m_B2 - m_B1 * m_B3 * m_B3 - m_B3 * m_B3 * m_B3 - m_B3 * m_B2 + m_B3;
103  m_MMatrix(2,2) = m_B3 * (m_B1 + m_B3 * m_B2);
104 
105  m_MMatrix /= (1 + m_B1 - m_B2 + m_B3) * (1 - m_B1 - m_B2 - m_B3) * (1 + m_B2 + (m_B1 - m_B3) * m_B3);
106 }
