568 typename DDIModelAveragingImageFilter<PixelScalarType>::TensorPixelType
569 DDIModelAveragingImageFilter<PixelScalarType>
570 ::getTensor(const VectorType &mu, const double kappa, const double diff, const double eaf)
571 {
572  double axialdiff = diff * (1.0 - 2.0 * eaf * anima::xi(kappa));
573  double radialdiff = diff * ((1.0 - eaf) / (kappa + 1.0) + eaf * anima::xi(kappa));
574 
575  TensorPixelType tensor;
576 
577  for (unsigned int i = 0;i < TensorPixelType::Dimension;++i)
578  {
579  for (unsigned int j = i;j < TensorPixelType::Dimension;++j)
580  {
581  double t = (axialdiff - radialdiff) * mu[i] * mu[j];
582 
583  tensor(i,j) = t;
584  tensor(j,i) = t;
585 
586  if (i == j)
587  tensor(i,j) += radialdiff;
588  }
589  }
590 
591  return tensor;
