58  std::vector <RealType> &diffs, std::vector <RealType> &scratch,
59  std::vector <RealType> &l_coefs, std::vector <RealType> &m_coefs,
60  std::vector <RealType> &r_coefs)
61  {
62  // Initialize m_0 and r_0
63  double delta = m_TimeStep * TInputImage::ImageDimension;
64 
65  l_coefs[0] = 0;
66  m_coefs[0] = 1.0 + 2 * delta * diffs[0] / m_SquareGradientDelta;
67  r_coefs[0] = - delta * (5 * diffs[0] - diffs[1]) / (4 * m_SquareGradientDelta);
68 
69  // Computing l, m and r coefs
70  unsigned int ln = data.size();
71 
72  for (unsigned int i = 1;i < ln - 1;++i)
73  {
74  l_coefs[i] = 1.0 / m_coefs[i-1];
75  l_coefs[i] *= - delta * (diffs[i] + (diffs[i+1] - diffs[i-1]) / 4.0) / m_SquareGradientDelta;
76 
77  m_coefs[i] = 1.0 + 2 * delta * diffs[i] / m_SquareGradientDelta;
78  m_coefs[i] -= l_coefs[i] * r_coefs[i-1];
79 
80  r_coefs[i] = - delta * (diffs[i] - (diffs[i+1] - diffs[i-1]) / 4.0) / m_SquareGradientDelta;
81  }
82 
83  // Compute las l, m, r coefs
84  l_coefs[ln-1] = 1.0 / m_coefs[ln-2];
85  l_coefs[ln-1] *= - delta * (5 * diffs[ln-1] - diffs[ln-2]) / (4.0 * m_SquareGradientDelta);
86 
87  m_coefs[ln-1] = 1 + 2 * delta * diffs[ln-1] / m_SquareGradientDelta;
88  m_coefs[ln-1] -= l_coefs[ln-1] * r_coefs[ln-2];
89 
90  r_coefs[ln-1] = 0;
91 
92  // Now compute forward substitution inside scratch
93  scratch[0] = data[0];
94  for (unsigned int i = 1;i < ln;++i)
95  scratch[i] = data[i] - l_coefs[i] * scratch[i-1];
96 
97  // And finally compute backward substitution inside outs
98  outs[ln-1] = scratch[ln-1] / m_coefs[ln-1];
99 
100  for (int i = ln-2;i >= 0;i--)
101  {
102  outs[i] = (scratch[i] - r_coefs[i] * outs[i+1]) / m_coefs[i];
103  }
104  }
